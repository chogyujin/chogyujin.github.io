https://www.acmicpc.net/problem/1463

역시 dp문제는 생각을 많이하게하는 것 같다. 꽤 많은 시간이 걸렸다. 

~~~
#include <iostream>
#include <cstdio>
using namespace std;
int d[1000001];
int temp;
int fun(int x) {
	if (x == 1) return 0;
	if (d[x] > 0) return d[x];
	d[x] = fun(x - 1) + 1;
	if (x % 3 == 0) {
		temp = fun(x / 3) + 1;
		if (d[x] > temp) d[x] = temp;
	}
	if (x % 2 == 0) {
		temp = fun(x / 2) + 1;
		if (d[x] > temp) d[x] = temp;
	}
	return d[x];
}
int main() {
	int x;
	
	scanf("%d", &x);
	fun(x);
	printf("%d\n",d[x]);
	/*
	cin >> x;
	cout << fun(x) << '\n';
    */
	return 0;
}
